## 4. –ö–æ–Ω—Ç—Ä–æ–ª—å —Ü–µ–ª–æ—Å—Ç–Ω–æ—Å—Ç–∏ –ø—Ä–æ—à–∏–≤–∫–∏ (CRC32)

### –ù–∞—Å—Ç—Ä–æ–π–∫–∞ —Å–∫—Ä–∏–ø—Ç–∞ –∫–æ–º–ø–æ–Ω–æ–≤—â–∏–∫–∞ (.ld)

**–†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏—è:** –†–∞—Å–ø–æ–ª–∞–≥–∞–π—Ç–µ —ç—Ç—É —Å–µ–∫—Ü–∏—é –≤ —Å–∞–º–æ–º **–∫–æ–Ω—Ü–µ –æ–±–ª–∞—Å—Ç–∏ FLASH**. –≠—Ç–æ –∫—Ä–∞–π–Ω–µ —É–¥–æ–±–Ω–æ –∫–∞–∫ –¥–ª—è —Å–∞–º–æ–≥–æ —Ä–∞–∑—Ä–∞–±–æ—Ç—á–∏–∫–∞, —Ç–∞–∫ –∏ –¥–ª—è —É—Ç–∏–ª–∏—Ç –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –ø–æ –≤–æ–∑–¥—É—Ö—É (OTA), –∑–∞–≥—Ä—É–∑—á–∏–∫–æ–≤ –∏ —Å–∫—Ä–∏–ø—Ç–æ–≤ –≤–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏–∏, —Ç–∞–∫ –∫–∞–∫ —Ä–∞–∑–º–µ—Ä –ø–æ–ª–µ–∑–Ω–æ–π –Ω–∞–≥—Ä—É–∑–∫–∏ (–∫–æ–¥–∞) –º–æ–∂–µ—Ç "–ø–ª–∞–≤–∞—Ç—å", –∞ –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–∞—è —Å—É–º–º–∞ –≤—Å–µ–≥–¥–∞ –±—É–¥–µ—Ç —á–µ—Ç–∫–æ –∑–∞–∫—Ä—ã–≤–∞—Ç—å –±–∏–Ω–∞—Ä–Ω—ã–π –æ–±—Ä–∞–∑.


### –ü—Ä–æ–≤–µ—Ä–∫–∞ CRC –≤ –∫–æ–¥–µ –º–∏–∫—Ä–æ–∫–æ–Ω—Ç—Ä–æ–ª–ª–µ—Ä–∞
–§—Ä–µ–π–º–≤–æ—Ä–∫ —Ç–æ–ª—å–∫–æ *–∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç* CRC –≤ —Ñ–∞–π–ª. –ü—Ä–æ–≤–µ—Ä–∫—É –ø—Ä–∏ –∑–∞–ø—É—Å–∫–µ –≤—ã –¥–æ–ª–∂–Ω—ã —Ä–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å —Å–∞–º–æ—Å—Ç–æ—è—Ç–µ–ª—å–Ω–æ. –ù–∏–∂–µ –ø—Ä–∏–≤–µ–¥–µ–Ω –ø—Ä–∏–º–µ—Ä –Ω–∞–¥–µ–∂–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –ø—Ä–æ–≤–µ—Ä–∫–∏, –∫–æ—Ç–æ—Ä—É—é —Å–ª–µ–¥—É–µ—Ç –≤—ã–∑—ã–≤–∞—Ç—å –≤ –Ω–∞—á–∞–ª–µ `main()` –ø–æ—Å–ª–µ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–∏ —Ç–∞–∫—Ç–∏—Ä–æ–≤–∞–Ω–∏—è –∏ UART:

```c

extern uint32_t __checksum_start[]; /// –ù–∞—á–∞–ª–æ –ø—Ä–æ–≤–µ—Ä—è–µ–º–æ–π –æ–±–ª–∞—Å—Ç–∏ (–∫–æ–Ω–µ—Ü –ø–æ–ª–µ–∑–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö)

extern uint32_t __checksum_size[];  /// –°–∏–º–≤–æ–ª, —á–µ–π –ê–î–†–ï–° —Ä–∞–≤–µ–Ω —Ä–∞–∑–º–µ—Ä—É –¥–∞–Ω–Ω—ã—Ö

/**
 * @brief  –í—ã–ø–æ–ª–Ω—è–µ—Ç –ø—Ä–æ–≤–µ—Ä–∫—É —Ü–µ–ª–æ—Å—Ç–Ω–æ—Å—Ç–∏ –ø—Ä–æ—à–∏–≤–∫–∏ (CRC32).
 * @param  hcrc –£–∫–∞–∑–∞—Ç–µ–ª—å –Ω–∞ —Ö–µ–Ω–¥–ª–µ—Ä CRC.
 * @return HAL_OK - —Ü–µ–ª–æ—Å—Ç–Ω–æ—Å—Ç—å –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∞, HAL_ERROR - –æ—à–∏–±–∫–∞.
 */
int CheckFirmwareIntegrity(CRC_HandleTypeDef *hcrc)
{
    // 1. –ü–æ–ª—É—á–∞–µ–º —Å—Ç–∞—Ä—Ç–æ–≤—ã–π –∞–¥—Ä–µ—Å
    uint32_t start_addr = ( uint32_t ) __checksum_start;

    // –í–∞–∂–Ω–æ: __checksum_size - —ç—Ç–æ —Å–∏–º–≤–æ–ª, –µ–≥–æ –∞–¥—Ä–µ—Å —Ä–∞–≤–µ–Ω —Ä–∞–∑–º–µ—Ä—É –≤ –±–∞–π—Ç–∞—Ö.

    // 2. –°—á–∏—Ç–∞–µ–º CRC –æ—Ç –Ω–∞—á–∞–ª–∞ –ø—Ä–æ—à–∏–≤–∫–∏ –¥–æ –Ω–∞—á–∞–ª–∞ —Å–µ–∫—Ü–∏–∏ .checksum
    uint32_t calc_crc = HAL_CRC_Calculate( hcrc, ( uint32_t* ) __checksum_start, data_len_words );

    // 3. –ß–∏—Ç–∞–µ–º —ç—Ç–∞–ª–æ–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –∑–∞–ø–∏—Å–∞–Ω–Ω–æ–µ —É—Ç–∏–ª–∏—Ç–æ–π —Å–±–æ—Ä–∫–∏.
    uint32_t stored_crc = __checksum_end;

    // 4. –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º.
    if ( calc_crc != stored_crc )
    {
        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –±–∞–∑–æ–≤—É—é –ø–µ—Ä–∏—Ñ–µ—Ä–∏—é –¥–ª—è –≤—ã–≤–æ–¥–∞ –æ—à–∏–±–∫–∏
        // MX_GPIO_Init();
        // MX_USART1_UART_Init();

        printf( "\n\n" );
        printf( "–ö–†–ò–¢–ò–ß–ï–°–ö–ê–Ø –û–®–ò–ë–ö–ê: –ü–†–û–í–ï–†–ö–ê –¶–ï–õ–û–°–¢–ù–û–°–¢–ò –ö–û–î–ê –ù–ï –ü–†–û–ô–î–ï–ù–ê!\n" );
        printf( "Start Addr: 0x%08lX\n", start_addr );
        printf( "Length:     %lu bytes (%lu words)\n", ( uint32_t ) __checksum_size, data_len_words );
        printf( "Stored CRC: 0x%08lX (at 0x%08lX)\n", stored_crc, ( uint32_t ) __checksum_end );
        printf( "Calc CRC:   0x%08lX\n", calc_crc );
        printf( "–°–∏—Å—Ç–µ–º–∞ –±—É–¥–µ—Ç –æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∞.\n" );

        return HAL_ERROR;
    }

    return HAL_OK;
}
```
